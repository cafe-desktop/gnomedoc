enum MapiAccess

typedef enum {
	MAPI_ACCESS_MODIFY            = (1 << 0),
	MAPI_ACCESS_READ              = (1 << 1),
	MAPI_ACCESS_DELETE            = (1 << 2),
	MAPI_ACCESS_CREATE_HIERARCHY  = (1 << 3),
	MAPI_ACCESS_CREATE_CONTENTS   = (1 << 4),
	MAPI_ACCESS_CREATE_ASSOCIATED = (1 << 5)

typedef enum {
